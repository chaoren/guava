015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.GwtIncompatible;
018import com.google.common.annotations.J2ktIncompatible;
019import javax.annotation.CheckForNull;
020
021/**
022 * Unchecked version of {@link java.util.concurrent.TimeoutException}.
023 *
024 * @author Kevin Bourrillion
025 * @since 1.0
026 */
027@J2ktIncompatible
028@GwtIncompatible
029public class UncheckedTimeoutException extends RuntimeException {
030  public UncheckedTimeoutException() {}
031
032  public UncheckedTimeoutException(@CheckForNull String message) {
033    super(message);
034  }
035
036  public UncheckedTimeoutException(@CheckForNull Throwable cause) {
037    super(cause);
038  }
039
040  public UncheckedTimeoutException(@CheckForNull String message, @CheckForNull Throwable cause) {
041    super(message, cause);
042  }
043
044  private static final long serialVersionUID = 0;
045}
