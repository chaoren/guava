015package com.google.common.reflect;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018
019import com.google.common.collect.ForwardingMap;
020import com.google.common.collect.ForwardingMapEntry;
021import com.google.common.collect.ForwardingSet;
022import com.google.common.collect.Iterators;
023import com.google.common.collect.Maps;
024import com.google.errorprone.annotations.CanIgnoreReturnValue;
025import com.google.errorprone.annotations.DoNotCall;
026import java.util.Iterator;
027import java.util.Map;
028import java.util.Set;
029import javax.annotation.CheckForNull;
030import org.checkerframework.checker.nullness.qual.NonNull;
031import org.checkerframework.checker.nullness.qual.Nullable;
032
033/**
034 * A mutable type-to-instance map. See also {@link ImmutableTypeToInstanceMap}.
035 *
036 * @author Ben Yu
037 * @since 13.0
038 */
039public final class MutableTypeToInstanceMap<B extends @Nullable Object>
040    extends ForwardingMap<TypeToken<? extends @NonNull B>, B> implements TypeToInstanceMap<B> {
041  /** Creates a new map. */
042  public MutableTypeToInstanceMap() {}
043
044  private final Map<TypeToken<? extends @NonNull B>, B> backingMap = Maps.newHashMap();
045
046  @Override
047  @CheckForNull
048  public <T extends @NonNull B> T getInstance(Class<T> type) {
049    return trustedGet(TypeToken.of(type));
050  }
051
052  @Override
053  @CheckForNull
054  public <T extends @NonNull B> T getInstance(TypeToken<T> type) {
055    return trustedGet(type.rejectTypeVariables());
056  }
057
058  @Override
059  @CanIgnoreReturnValue
060  @CheckForNull
061  public <T extends B> T putInstance(Class<@NonNull T> type, @ParametricNullness T value) {
062    return trustedPut(TypeToken.of(type), value);
063  }
064
065  @Override
066  @CanIgnoreReturnValue
067  @CheckForNull
068  public <T extends B> T putInstance(TypeToken<@NonNull T> type, @ParametricNullness T value) {
069    return this.<T>trustedPut(type.rejectTypeVariables(), value);
070  }
071
072  /**
073   * Not supported. Use {@link #putInstance} instead.
074   *
075   * @deprecated unsupported operation
076   * @throws UnsupportedOperationException always
077   */
078  @CanIgnoreReturnValue
079  @Deprecated
080  @Override
081  @DoNotCall("Always throws UnsupportedOperationException")
082  @CheckForNull
083  public B put(TypeToken<? extends @NonNull B> key, @ParametricNullness B value) {
084    throw new UnsupportedOperationException("Please use putInstance() instead.");
085  }
086
087  /**
088   * Not supported. Use {@link #putInstance} instead.
089   *
090   * @deprecated unsupported operation
091   * @throws UnsupportedOperationException always
092   */
093  @Deprecated
094  @Override
095  @DoNotCall("Always throws UnsupportedOperationException")
096  public void putAll(Map<? extends TypeToken<? extends @NonNull B>, ? extends B> map) {
097    throw new UnsupportedOperationException("Please use putInstance() instead.");
098  }
099
100  @Override
101  public Set<Entry<TypeToken<? extends @NonNull B>, B>> entrySet() {
102    return UnmodifiableEntry.transformEntries(super.entrySet());
103  }
104
105  @Override
106  protected Map<TypeToken<? extends @NonNull B>, B> delegate() {
107    return backingMap;
108  }
109
110  @SuppressWarnings("unchecked") // value could not get in if not a T
111  @CheckForNull
112  private <T extends B> T trustedPut(TypeToken<@NonNull T> type, @ParametricNullness T value) {
113    return (T) backingMap.put(type, value);
114  }
115
116  @SuppressWarnings("unchecked") // value could not get in if not a T
117  @CheckForNull
118  private <T extends @NonNull B> T trustedGet(TypeToken<T> type) {
119    return (T) backingMap.get(type);
120  }
121
122  private static final class UnmodifiableEntry<K, V extends @Nullable Object>
123      extends ForwardingMapEntry<K, V> {
124
125    private final Entry<K, V> delegate;
126
127    static <K, V extends @Nullable Object> Set<Entry<K, V>> transformEntries(
128        Set<Entry<K, V>> entries) {
129      return new ForwardingSet<Map.Entry<K, V>>() {
130        @Override
131        protected Set<Entry<K, V>> delegate() {
132          return entries;
133        }
134
135        @Override
136        public Iterator<Entry<K, V>> iterator() {
137          return UnmodifiableEntry.transformEntries(super.iterator());
138        }
139
140        @Override
141        public Object[] toArray() {
142          /*
143           * standardToArray returns `@Nullable Object[]` rather than `Object[]` but only because it
144           * can be used with collections that may contain null. This collection is a collection of
145           * non-null Entry objects (Entry objects that might contain null values but are not
146           * themselves null), so we can treat it as a plain `Object[]`.
147           */
148          @SuppressWarnings("nullness")
149          Object[] result = standardToArray();
150          return result;
151        }
152
153        @Override
154        @SuppressWarnings("nullness") // b/192354773 in our checker affects toArray declarations
155        public <T extends @Nullable Object> T[] toArray(T[] array) {
156          return standardToArray(array);
157        }
158      };
159    }
160
161    private static <K, V extends @Nullable Object> Iterator<Entry<K, V>> transformEntries(
162        Iterator<Entry<K, V>> entries) {
163      return Iterators.transform(entries, UnmodifiableEntry::new);
164    }
165
166    private UnmodifiableEntry(Entry<K, V> delegate) {
167      this.delegate = checkNotNull(delegate);
168    }
169
170    @Override
171    protected Entry<K, V> delegate() {
172      return delegate;
173    }
174
175    @Override
176    @ParametricNullness
177    public V setValue(@ParametricNullness V value) {
178      throw new UnsupportedOperationException();
179    }
180  }
181}
