017package com.google.common.collect;
018
019import com.google.common.annotations.GwtIncompatible;
020import com.google.common.annotations.J2ktIncompatible;
021import com.google.errorprone.annotations.DoNotMock;
022
023/**
024 * Provides similar behavior to {@link String#intern} for any immutable type. Common implementations
025 * are available from the {@link Interners} class.
026 *
027 * <p>Note that {@code String.intern()} has some well-known performance limitations, and should
028 * generally be avoided. Prefer {@link Interners#newWeakInterner} or another {@code Interner}
029 * implementation even for {@code String} interning.
030 *
031 * @author Kevin Bourrillion
032 * @since 3.0
033 */
034@DoNotMock("Use Interners.new*Interner")
035@J2ktIncompatible
036@GwtIncompatible
037public interface Interner<E> {
038  /**
039   * Chooses and returns the representative instance for any of a collection of instances that are
040   * equal to each other. If two {@linkplain Object#equals equal} inputs are given to this method,
041   * both calls will return the same instance. That is, {@code intern(a).equals(a)} always holds,
042   * and {@code intern(a) == intern(b)} if and only if {@code a.equals(b)}. Note that {@code
043   * intern(a)} is permitted to return one instance now and a different instance later if the
044   * original interned instance was garbage-collected.
045   *
046   * <p><b>Warning:</b> do not use with mutable objects.
047   *
048   * @throws NullPointerException if {@code sample} is null
049   */
050  E intern(E sample);
051}
