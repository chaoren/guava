017package com.google.common.collect;
018
019import com.google.common.annotations.GwtCompatible;
020import java.util.Comparator;
021import java.util.SortedSet;
022import javax.annotation.CheckForNull;
023import org.checkerframework.checker.nullness.qual.Nullable;
024
025/**
026 * A sorted set multimap which forwards all its method calls to another sorted set multimap.
027 * Subclasses should override one or more methods to modify the behavior of the backing multimap as
028 * desired per the <a href="http://en.wikipedia.org/wiki/Decorator_pattern">decorator pattern</a>.
029 *
030 * <p><b>{@code default} method warning:</b> This class does <i>not</i> forward calls to {@code
031 * default} methods. Instead, it inherits their default implementations. When those implementations
032 * invoke methods, they invoke methods on the {@code ForwardingSortedSetMultimap}.
033 *
034 * @author Kurt Alfred Kluever
035 * @since 3.0
036 */
037@GwtCompatible
038public abstract class ForwardingSortedSetMultimap<
039        K extends @Nullable Object, V extends @Nullable Object>
040    extends ForwardingSetMultimap<K, V> implements SortedSetMultimap<K, V> {
041
042  /** Constructor for use by subclasses. */
043  protected ForwardingSortedSetMultimap() {}
044
045  @Override
046  protected abstract SortedSetMultimap<K, V> delegate();
047
048  @Override
049  public SortedSet<V> get(@ParametricNullness K key) {
050    return delegate().get(key);
051  }
052
053  @Override
054  public SortedSet<V> removeAll(@CheckForNull Object key) {
055    return delegate().removeAll(key);
056  }
057
058  @Override
059  public SortedSet<V> replaceValues(@ParametricNullness K key, Iterable<? extends V> values) {
060    return delegate().replaceValues(key, values);
061  }
062
063  @Override
064  @CheckForNull
065  public Comparator<? super V> valueComparator() {
066    return delegate().valueComparator();
067  }
068}
