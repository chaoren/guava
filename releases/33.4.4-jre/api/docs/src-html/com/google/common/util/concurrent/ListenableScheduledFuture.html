015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.GwtCompatible;
018import java.util.concurrent.ScheduledFuture;
019import org.jspecify.annotations.Nullable;
020
021/**
022 * Helper interface to implement both {@link ListenableFuture} and {@link ScheduledFuture}.
023 *
024 * @author Anthony Zana
025 * @since 15.0
026 */
027@GwtCompatible
028public interface ListenableScheduledFuture<V extends @Nullable Object>
029    extends ScheduledFuture<V>, ListenableFuture<V> {}
