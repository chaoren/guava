015package com.google.common.io;
016
017import com.google.common.annotations.GwtIncompatible;
018import com.google.common.annotations.J2ktIncompatible;
019import com.google.errorprone.annotations.CanIgnoreReturnValue;
020import java.io.IOException;
021import org.jspecify.annotations.Nullable;
022
023/**
024 * A callback to be used with the streaming {@code readLines} methods.
025 *
026 * <p>{@link #processLine} will be called for each line that is read, and should return {@code
027 * false} when you want to stop processing.
028 *
029 * @author Miles Barr
030 * @since 1.0
031 */
032@J2ktIncompatible
033@GwtIncompatible
034public interface LineProcessor<T extends @Nullable Object> {
035
036  /**
037   * This method will be called once for each line.
038   *
039   * @param line the line read from the input, without delimiter
040   * @return true to continue processing, false to stop
041   */
042  @CanIgnoreReturnValue // some uses know that their processor never returns false
043  boolean processLine(String line) throws IOException;
044
045  /** Return the result of processing all the lines. */
046  @ParametricNullness
047  T getResult();
048}
