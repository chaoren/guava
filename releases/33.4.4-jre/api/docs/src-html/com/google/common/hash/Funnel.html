015package com.google.common.hash;
016
017import com.google.common.annotations.Beta;
018import com.google.errorprone.annotations.DoNotMock;
019import java.io.Serializable;
020import org.jspecify.annotations.Nullable;
021
022/**
023 * An object which can send data from an object of type {@code T} into a {@code PrimitiveSink}.
024 * Implementations for common types can be found in {@link Funnels}.
025 *
026 * <p>Note that serialization of {@linkplain BloomFilter bloom filters} requires the proper
027 * serialization of funnels. When possible, it is recommended that funnels be implemented as a
028 * single-element enum to maintain serialization guarantees. See Effective Java (2nd Edition), Item
029 * 3: "Enforce the singleton property with a private constructor or an enum type". For example:
030 *
031 * <pre>{@code
032 * public enum PersonFunnel implements Funnel<Person> {
033 *   INSTANCE;
034 *   public void funnel(Person person, PrimitiveSink into) {
035 *     into.putUnencodedChars(person.getFirstName())
036 *         .putUnencodedChars(person.getLastName())
037 *         .putInt(person.getAge());
038 *   }
039 * }
040 * }</pre>
041 *
042 * @author Dimitris Andreou
043 * @since 11.0
044 */
045@Beta
046@DoNotMock("Implement with a lambda")
047public interface Funnel<T extends @Nullable Object> extends Serializable {
048
049  /**
050   * Sends a stream of data from the {@code from} object into the sink {@code into}. There is no
051   * requirement that this data be complete enough to fully reconstitute the object later.
052   *
053   * @since 12.0 (in Guava 11.0, {@code PrimitiveSink} was named {@code Sink})
054   */
055  void funnel(@ParametricNullness T from, PrimitiveSink into);
056}
