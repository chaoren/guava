017package com.google.common.collect;
018
019import com.google.common.annotations.GwtCompatible;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021import com.google.errorprone.annotations.DoNotMock;
022import java.util.Iterator;
023import java.util.NoSuchElementException;
024import org.checkerframework.checker.nullness.qual.Nullable;
025
026/**
027 * An iterator that supports a one-element lookahead while iterating.
028 *
029 * <p>See the Guava User Guide article on <a href=
030 * "https://github.com/google/guava/wiki/CollectionHelpersExplained#peekingiterator">{@code
031 * PeekingIterator}</a>.
032 *
033 * @author Mick Killianey
034 * @since 2.0
035 */
036@DoNotMock("Use Iterators.peekingIterator")
037@GwtCompatible
038public interface PeekingIterator<E extends @Nullable Object> extends Iterator<E> {
039  /**
040   * Returns the next element in the iteration, without advancing the iteration.
041   *
042   * <p>Calls to {@code peek()} should not change the state of the iteration, except that it
043   * <i>may</i> prevent removal of the most recent element via {@link #remove()}.
044   *
045   * @throws NoSuchElementException if the iteration has no more elements according to {@link
046   *     #hasNext()}
047   */
048  @ParametricNullness
049  E peek();
050
051  /**
052   * {@inheritDoc}
053   *
054   * <p>The objects returned by consecutive calls to {@link #peek()} then {@link #next()} are
055   * guaranteed to be equal to each other.
056   */
057  @CanIgnoreReturnValue
058  @Override
059  @ParametricNullness
060  E next();
061
062  /**
063   * {@inheritDoc}
064   *
065   * <p>Implementations may or may not support removal when a call to {@link #peek()} has occurred
066   * since the most recent call to {@link #next()}.
067   *
068   * @throws IllegalStateException if there has been a call to {@link #peek()} since the most recent
069   *     call to {@link #next()} and this implementation does not support this sequence of calls
070   *     (optional)
071   */
072  @Override
073  void remove();
074}
