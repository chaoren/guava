043/**
044 * A Bloom filter for instances of {@code T}. A Bloom filter offers an approximate containment test
045 * with one-sided error: if it claims that an element is contained in it, this might be in error,
046 * but if it claims that an element is <i>not</i> contained in it, then this is definitely true.
047 *
048 * <p>If you are unfamiliar with Bloom filters, this nice <a
049 * href="http://llimllib.github.io/bloomfilter-tutorial/">tutorial</a> may help you understand how
050 * they work.
051 *
052 * <p>The false positive probability ({@code FPP}) of a Bloom filter is defined as the probability
053 * that {@linkplain #mightContain(Object)} will erroneously return {@code true} for an object that
054 * has not actually been put in the {@code BloomFilter}.
055 *
056 * <p>Bloom filters are serializable. They also support a more compact serial representation via the
057 * {@link #writeTo} and {@link #readFrom} methods. Both serialized forms will continue to be
058 * supported by future versions of this library. However, serial forms generated by newer versions
059 * of the code may not be readable by older versions of the code (e.g., a serialized Bloom filter
060 * generated today may <i>not</i> be readable by a binary that was compiled 6 months ago).
061 *
062 * <p>As of Guava 23.0, this class is thread-safe and lock-free. It internally uses atomics and
063 * compare-and-swap to ensure correctness when multiple threads are used to access it.
064 *
065 * @param <T> the type of instances that the {@code BloomFilter} accepts
066 * @author Dimitris Andreou
067 * @author Kevin Bourrillion
068 * @since 11.0 (thread-safe since 23.0)
069 */
070@Beta
071public final class BloomFilter<T extends @Nullable Object> implements Predicate<T>, Serializable {
072  /**
073   * A strategy to translate T instances, to {@code numHashFunctions} bit indexes.
074   *
075   * <p>Implementations should be collections of pure functions (i.e. stateless).
076   */
077  interface Strategy extends java.io.Serializable {
078
079    /**
080     * Sets {@code numHashFunctions} bits of the given bit array, by hashing a user element.
081     *
082     * <p>Returns whether any bits changed as a result of this operation.
083     */
084    <T extends @Nullable Object> boolean put(
085        @ParametricNullness T object,
086        Funnel<? super T> funnel,
087        int numHashFunctions,
088        LockFreeBitArray bits);
089
090    /**
091     * Queries {@code numHashFunctions} bits of the given bit array, by hashing a user element;
092     * returns {@code true} if and only if all selected bits are set.
093     */
094    <T extends @Nullable Object> boolean mightContain(
095        @ParametricNullness T object,
096        Funnel<? super T> funnel,
097        int numHashFunctions,
098        LockFreeBitArray bits);
099
100    /**
101     * Identifier used to encode this strategy, when marshalled as part of a BloomFilter. Only
102     * values in the [-128, 127] range are valid for the compact serial form. Non-negative values
103     * are reserved for enums defined in BloomFilterStrategies; negative values are reserved for any
104     * custom, stateful strategy we may define (e.g. any kind of strategy that would depend on user
105     * input).
106     */
107    int ordinal();
108  }

110  /** The bit set of the BloomFilter (not necessarily power of 2!) */
111  private final LockFreeBitArray bits;
112
113  /** Number of hashes per element */
114  private final int numHashFunctions;
115
116  /** The funnel to translate Ts to bytes */
117  private final Funnel<? super T> funnel;
118
119  /** The strategy we employ to map an element T to {@code numHashFunctions} bit indexes. */
120  private final Strategy strategy;

122  /** Natural logarithm of 2, used to optimize calculations in Bloom filter sizing. */
123  private static final double LOG_TWO = Math.log(2);
124
125  /** Square of the natural logarithm of 2, reused to optimize the bit size calculation. */
126  private static final double SQUARED_LOG_TWO = LOG_TWO * LOG_TWO;

128  /** Creates a BloomFilter. */
129  private BloomFilter(
130      LockFreeBitArray bits, int numHashFunctions, Funnel<? super T> funnel, Strategy strategy) {
131    checkArgument(numHashFunctions > 0, "numHashFunctions (%s) must be > 0", numHashFunctions);
132    checkArgument(
133        numHashFunctions <= 255, "numHashFunctions (%s) must be <= 255", numHashFunctions);
134    this.bits = checkNotNull(bits);
135    this.numHashFunctions = numHashFunctions;
136    this.funnel = checkNotNull(funnel);
137    this.strategy = checkNotNull(strategy);
138  }

168  /**
169   * Puts an element into this {@code BloomFilter}. Ensures that subsequent invocations of {@link
170   * #mightContain(Object)} with the same element will always return {@code true}.
171   *
172   * @return true if the Bloom filter's bits changed as a result of this operation. If the bits
173   *     changed, this is <i>definitely</i> the first time {@code object} has been added to the
174   *     filter. If the bits haven't changed, this <i>might</i> be the first time {@code object} has
175   *     been added to the filter. Note that {@code put(t)} always returns the <i>opposite</i>
176   *     result to what {@code mightContain(t)} would have returned at the time it is called.
177   * @since 12.0 (present in 11.0 with {@code void} return type})
178   */
179  @CanIgnoreReturnValue
180  public boolean put(@ParametricNullness T object) {
181    return strategy.put(object, funnel, numHashFunctions, bits);
182  }

184  /**
185   * Returns the probability that {@linkplain #mightContain(Object)} will erroneously return {@code
186   * true} for an object that has not actually been put in the {@code BloomFilter}.
187   *
188   * <p>Ideally, this number should be close to the {@code fpp} parameter passed in {@linkplain
189   * #create(Funnel, int, double)}, or smaller. If it is significantly higher, it is usually the
190   * case that too many elements (more than expected) have been put in the {@code BloomFilter},
191   * degenerating it.
192   *
193   * @since 14.0 (since 11.0 as expectedFalsePositiveProbability())
194   */
195  public double expectedFpp() {
196    return Math.pow((double) bits.bitCount() / bitSize(), numHashFunctions);
197  }
198
199  /**
200   * Returns an estimate for the total number of distinct elements that have been added to this
201   * Bloom filter. This approximation is reasonably accurate if it does not exceed the value of
202   * {@code expectedInsertions} that was used when constructing the filter.
203   *
204   * @since 22.0
205   */
206  public long approximateElementCount() {
207    long bitSize = bits.bitSize();
208    long bitCount = bits.bitCount();
209
210    /**
211     * Each insertion is expected to reduce the # of clear bits by a factor of
212     * `numHashFunctions/bitSize`. So, after n insertions, expected bitCount is `bitSize * (1 - (1 -
213     * numHashFunctions/bitSize)^n)`. Solving that for n, and approximating `ln x` as `x - 1` when x
214     * is close to 1 (why?), gives the following formula.
215     */
216    double fractionOfBitsSet = (double) bitCount / bitSize;
217    return DoubleMath.roundToLong(
218        -Math.log1p(-fractionOfBitsSet) * bitSize / numHashFunctions, RoundingMode.HALF_UP);
219  }

227  /**
228   * Determines whether a given Bloom filter is compatible with this Bloom filter. For two Bloom
229   * filters to be compatible, they must:
230   *
231   * <ul>
232   *   <li>not be the same instance
233   *   <li>have the same number of hash functions
234   *   <li>have the same bit size
235   *   <li>have the same strategy
236   *   <li>have equal funnels
237   * </ul>
238   *
239   * @param that The Bloom filter to check for compatibility.
240   * @since 15.0
241   */
242  public boolean isCompatible(BloomFilter<T> that) {
243    checkNotNull(that);
244    return this != that
245        && this.numHashFunctions == that.numHashFunctions
246        && this.bitSize() == that.bitSize()
247        && this.strategy.equals(that.strategy)
248        && this.funnel.equals(that.funnel);
249  }
250
251  /**
252   * Combines this Bloom filter with another Bloom filter by performing a bitwise OR of the
253   * underlying data. The mutations happen to <b>this</b> instance. Callers must ensure the Bloom
254   * filters are appropriately sized to avoid saturating them.
255   *
256   * @param that The Bloom filter to combine this Bloom filter with. It is not mutated.
257   * @throws IllegalArgumentException if {@code isCompatible(that) == false}
258   * @since 15.0
259   */
260  public void putAll(BloomFilter<T> that) {
261    checkNotNull(that);
262    checkArgument(this != that, "Cannot combine a BloomFilter with itself.");
263    checkArgument(
264        this.numHashFunctions == that.numHashFunctions,
265        "BloomFilters must have the same number of hash functions (%s != %s)",
266        this.numHashFunctions,
267        that.numHashFunctions);
268    checkArgument(
269        this.bitSize() == that.bitSize(),
270        "BloomFilters must have the same size underlying bit arrays (%s != %s)",
271        this.bitSize(),
272        that.bitSize());
273    checkArgument(
274        this.strategy.equals(that.strategy),
275        "BloomFilters must have equal strategies (%s != %s)",
276        this.strategy,
277        that.strategy);
278    checkArgument(
279        this.funnel.equals(that.funnel),
280        "BloomFilters must have equal funnels (%s != %s)",
281        this.funnel,
282        that.funnel);
283    this.bits.putAll(that.bits);
284  }

286  @Override
287  public boolean equals(@Nullable Object object) {
288    if (object == this) {
289      return true;
290    }
291    if (object instanceof BloomFilter) {
292      BloomFilter<?> that = (BloomFilter<?>) object;
293      return this.numHashFunctions == that.numHashFunctions
294          && this.funnel.equals(that.funnel)
295          && this.bits.equals(that.bits)
296          && this.strategy.equals(that.strategy);
297    }
298    return false;
299  }

332  /**
333   * Returns a {@code Collector} expecting the specified number of insertions, and yielding a {@link
334   * BloomFilter} with the specified expected false positive probability.
335   *
336   * <p>Note that if the {@code Collector} receives significantly more elements than specified, the
337   * resulting {@code BloomFilter} will suffer a sharp deterioration of its false positive
338   * probability.
339   *
340   * <p>The constructed {@code BloomFilter} will be serializable if the provided {@code Funnel<T>}
341   * is.
342   *
343   * <p>It is recommended that the funnel be implemented as a Java enum. This has the benefit of
344   * ensuring proper serialization and deserialization, which is important since {@link #equals}
345   * also relies on object identity of funnels.
346   *
347   * @param funnel the funnel of T's that the constructed {@code BloomFilter} will use
348   * @param expectedInsertions the number of expected insertions to the constructed {@code
349   *     BloomFilter}; must be positive
350   * @param fpp the desired false positive probability (must be positive and less than 1.0)
351   * @return a {@code Collector} generating a {@code BloomFilter} of the received elements
352   * @since 23.0 (but only since 33.4.0 in the Android flavor)
353   */
354  public static <T extends @Nullable Object> Collector<T, ?, BloomFilter<T>> toBloomFilter(
355      Funnel<? super T> funnel, long expectedInsertions, double fpp) {
356    checkNotNull(funnel);
357    checkArgument(
358        expectedInsertions >= 0, "Expected insertions (%s) must be >= 0", expectedInsertions);
359    checkArgument(fpp > 0.0, "False positive probability (%s) must be > 0.0", fpp);
360    checkArgument(fpp < 1.0, "False positive probability (%s) must be < 1.0", fpp);
361    return Collector.of(
362        () -> BloomFilter.create(funnel, expectedInsertions, fpp),
363        BloomFilter::put,
364        (bf1, bf2) -> {
365          bf1.putAll(bf2);
366          return bf1;
367        },
368        Collector.Characteristics.UNORDERED,
369        Collector.Characteristics.CONCURRENT);
370  }

423  @VisibleForTesting
424  static <T extends @Nullable Object> BloomFilter<T> create(
425      Funnel<? super T> funnel, long expectedInsertions, double fpp, Strategy strategy) {
426    checkNotNull(funnel);
427    checkArgument(
428        expectedInsertions >= 0, "Expected insertions (%s) must be >= 0", expectedInsertions);
429    checkArgument(fpp > 0.0, "False positive probability (%s) must be > 0.0", fpp);
430    checkArgument(fpp < 1.0, "False positive probability (%s) must be < 1.0", fpp);
431    checkNotNull(strategy);
432
433    if (expectedInsertions == 0) {
434      expectedInsertions = 1;
435    }
436    /*
437     * TODO(user): Put a warning in the javadoc about tiny fpp values, since the resulting size
438     * is proportional to -log(p), but there is not much of a point after all, e.g.
439     * optimalM(1000, 0.0000000000000001) = 76680 which is less than 10kb. Who cares!
440     */
441    long numBits = optimalNumOfBits(expectedInsertions, fpp);
442    int numHashFunctions = optimalNumOfHashFunctions(fpp);
443    try {
444      return new BloomFilter<>(new LockFreeBitArray(numBits), numHashFunctions, funnel, strategy);
445    } catch (IllegalArgumentException e) {
446      throw new IllegalArgumentException("Could not create BloomFilter of " + numBits + " bits", e);
447    }
448  }

499  // Cheat sheet:
500  //
501  // m: total bits
502  // n: expected insertions
503  // b: m/n, bits per insertion
504  // p: expected false positive probability
505  //
506  // 1) Optimal k = b * ln2
507  // 2) p = (1 - e ^ (-kn/m))^k
508  // 3) For optimal k: p = 2 ^ (-k) ~= 0.6185^b
509  // 4) For optimal k: m = -nlnp / ((ln2) ^ 2)
510
511  /**
512   * Computes the optimal number of hash functions (k) for a given false positive probability (p).
513   *
514   * <p>See http://en.wikipedia.org/wiki/File:Bloom_filter_fp_probability.svg for the formula.
515   *
516   * @param p desired false positive probability (must be between 0 and 1, exclusive)
517   */
518  @VisibleForTesting
519  static int optimalNumOfHashFunctions(double p) {
520    // -log(p) / log(2), ensuring the result is rounded to avoid truncation.
521    return max(1, (int) Math.round(-Math.log(p) / LOG_TWO));
522  }
523
524  /**
525   * Computes m (total bits of Bloom filter) which is expected to achieve, for the specified
526   * expected insertions, the required false positive probability.
527   *
528   * <p>See http://en.wikipedia.org/wiki/Bloom_filter#Probability_of_false_positives for the
529   * formula.
530   *
531   * @param n expected insertions (must be positive)
532   * @param p false positive rate (must be 0 < p < 1)
533   */
534  @VisibleForTesting
535  static long optimalNumOfBits(long n, double p) {
536    if (p == 0) {
537      p = Double.MIN_VALUE;
538    }
539    return (long) (-n * Math.log(p) / SQUARED_LOG_TWO);
540  }

550  private static class SerialForm<T extends @Nullable Object> implements Serializable {
551    final long[] data;
552    final int numHashFunctions;
553    final Funnel<? super T> funnel;
554    final Strategy strategy;
555
556    SerialForm(BloomFilter<T> bf) {
557      this.data = LockFreeBitArray.toPlainArray(bf.bits.data);
558      this.numHashFunctions = bf.numHashFunctions;
559      this.funnel = bf.funnel;
560      this.strategy = bf.strategy;
561    }
562
563    Object readResolve() {
564      return new BloomFilter<T>(new LockFreeBitArray(data), numHashFunctions, funnel, strategy);
565    }

567    private static final long serialVersionUID = 1;
568  }

570  /**
571   * Writes this {@code BloomFilter} to an output stream, with a custom format (not Java
572   * serialization). This has been measured to save at least 400 bytes compared to regular
573   * serialization.
574   *
575   * <p>Use {@linkplain #readFrom(InputStream, Funnel)} to reconstruct the written BloomFilter.
576   */
577  public void writeTo(OutputStream out) throws IOException {
578    // Serial form:
579    // 1 signed byte for the strategy
580    // 1 unsigned byte for the number of hash functions
581    // 1 big endian int, the number of longs in our bitset
582    // N big endian longs of our bitset
583    DataOutputStream dout = new DataOutputStream(out);
584    dout.writeByte(SignedBytes.checkedCast(strategy.ordinal()));
585    dout.writeByte(UnsignedBytes.checkedCast(numHashFunctions)); // note: checked at the c'tor
586    dout.writeInt(bits.data.length());
587    for (int i = 0; i < bits.data.length(); i++) {
588      dout.writeLong(bits.data.get(i));
589    }
590  }

592  /**
593   * Reads a byte stream, which was written by {@linkplain #writeTo(OutputStream)}, into a {@code
594   * BloomFilter}.
595   *
596   * <p>The {@code Funnel} to be used is not encoded in the stream, so it must be provided here.
597   * <b>Warning:</b> the funnel provided <b>must</b> behave identically to the one used to populate
598   * the original Bloom filter!
599   *
600   * @throws IOException if the InputStream throws an {@code IOException}, or if its data does not
601   *     appear to be a BloomFilter serialized using the {@linkplain #writeTo(OutputStream)} method.
602   */
603  @SuppressWarnings("CatchingUnchecked") // sneaky checked exception
604  public static <T extends @Nullable Object> BloomFilter<T> readFrom(
605      InputStream in, Funnel<? super T> funnel) throws IOException {
606    checkNotNull(in, "InputStream");
607    checkNotNull(funnel, "Funnel");
608    int strategyOrdinal = -1;
609    int numHashFunctions = -1;
610    int dataLength = -1;
611    try {
612      DataInputStream din = new DataInputStream(in);
613      // currently this assumes there is no negative ordinal; will have to be updated if we
614      // add non-stateless strategies (for which we've reserved negative ordinals; see
615      // Strategy.ordinal()).
616      strategyOrdinal = din.readByte();
617      numHashFunctions = UnsignedBytes.toInt(din.readByte());
618      dataLength = din.readInt();
619
620      Strategy strategy = BloomFilterStrategies.values()[strategyOrdinal];
621
622      LockFreeBitArray dataArray = new LockFreeBitArray(LongMath.checkedMultiply(dataLength, 64L));
623      for (int i = 0; i < dataLength; i++) {
624        dataArray.putData(i, din.readLong());
625      }
626
627      return new BloomFilter<>(dataArray, numHashFunctions, funnel, strategy);
628    } catch (IOException e) {
629      throw e;
630    } catch (Exception e) { // sneaky checked exception
631      String message =
632          "Unable to deserialize BloomFilter from InputStream."
633              + " strategyOrdinal: "
634              + strategyOrdinal
635              + " numHashFunctions: "
636              + numHashFunctions
637              + " dataLength: "
638              + dataLength;
639      throw new IOException(message, e);
640    }
641  }
