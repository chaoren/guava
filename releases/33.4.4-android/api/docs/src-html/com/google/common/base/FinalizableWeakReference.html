015package com.google.common.base;
016
017import com.google.common.annotations.GwtIncompatible;
018import com.google.common.annotations.J2ktIncompatible;
019import java.lang.ref.ReferenceQueue;
020import java.lang.ref.WeakReference;
021import org.jspecify.annotations.Nullable;
022
023/**
024 * Weak reference with a {@code finalizeReferent()} method which a background thread invokes after
025 * the garbage collector reclaims the referent. This is a simpler alternative to using a {@link
026 * ReferenceQueue}.
027 *
028 * @author Bob Lee
029 * @since 2.0
030 */
031@J2ktIncompatible
032@GwtIncompatible
033public abstract class FinalizableWeakReference<T> extends WeakReference<T>
034    implements FinalizableReference {
035  /**
036   * Constructs a new finalizable weak reference.
037   *
038   * @param referent to weakly reference
039   * @param queue that should finalize the referent
040   */
041  protected FinalizableWeakReference(@Nullable T referent, FinalizableReferenceQueue queue) {
042    super(referent, queue.queue);
043    queue.cleanUp();
044  }
045}
