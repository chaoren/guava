015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.GwtIncompatible;
018import com.google.errorprone.annotations.DoNotMock;
019import java.util.Collection;
020import java.util.List;
021import java.util.concurrent.Callable;
022import java.util.concurrent.ExecutorService;
023import java.util.concurrent.Future;
024import java.util.concurrent.RejectedExecutionException;
025import java.util.concurrent.TimeUnit;
026import org.jspecify.annotations.Nullable;
027
028/**
029 * An {@link ExecutorService} that returns {@link ListenableFuture} instances. To create an instance
030 * from an existing {@link ExecutorService}, call {@link
031 * MoreExecutors#listeningDecorator(ExecutorService)}.
032 *
033 * @author Chris Povirk
034 * @since 10.0
035 */
036@DoNotMock(
037    "Use TestingExecutors.sameThreadScheduledExecutor, or wrap a real Executor from "
038        + "java.util.concurrent.Executors with MoreExecutors.listeningDecorator")
039@GwtIncompatible
040public interface ListeningExecutorService extends ExecutorService {
041  /**
042   * @return a {@code ListenableFuture} representing pending completion of the task
043   * @throws RejectedExecutionException {@inheritDoc}
044   */
045  @Override
046  <T extends @Nullable Object> ListenableFuture<T> submit(Callable<T> task);
047
048  /**
049   * @return a {@code ListenableFuture} representing pending completion of the task
050   * @throws RejectedExecutionException {@inheritDoc}
051   */
052  @Override
053  ListenableFuture<?> submit(Runnable task);
054
055  /**
056   * @return a {@code ListenableFuture} representing pending completion of the task
057   * @throws RejectedExecutionException {@inheritDoc}
058   */
059  @Override
060  <T extends @Nullable Object> ListenableFuture<T> submit(
061      Runnable task, @ParametricNullness T result);
062
063  /**
064   * {@inheritDoc}
065   *
066   * <p>All elements in the returned list must be {@link ListenableFuture} instances. The easiest
067   * way to obtain a {@code List<ListenableFuture<T>>} from this method is an unchecked (but safe)
068   * cast:
069   *
070   * <pre>
071   *   {@code @SuppressWarnings("unchecked") // guaranteed by invokeAll contract}
072   *   {@code List<ListenableFuture<T>> futures = (List) executor.invokeAll(tasks);}
073   * </pre>
074   *
075   * @return A list of {@code ListenableFuture} instances representing the tasks, in the same
076   *     sequential order as produced by the iterator for the given task list, each of which has
077   *     completed.
078   * @throws RejectedExecutionException {@inheritDoc}
079   * @throws NullPointerException if any task is null
080   */
081  @Override
082  <T extends @Nullable Object> List<Future<T>> invokeAll(Collection<? extends Callable<T>> tasks)
083      throws InterruptedException;
084
085  /**
086   * {@inheritDoc}
087   *
088   * <p>All elements in the returned list must be {@link ListenableFuture} instances. The easiest
089   * way to obtain a {@code List<ListenableFuture<T>>} from this method is an unchecked (but safe)
090   * cast:
091   *
092   * <pre>
093   *   {@code @SuppressWarnings("unchecked") // guaranteed by invokeAll contract}
094   *   {@code List<ListenableFuture<T>> futures = (List) executor.invokeAll(tasks, timeout, unit);}
095   * </pre>
096   *
097   * @return a list of {@code ListenableFuture} instances representing the tasks, in the same
098   *     sequential order as produced by the iterator for the given task list. If the operation did
099   *     not time out, each task will have completed. If it did time out, some of these tasks will
100   *     not have completed.
101   * @throws RejectedExecutionException {@inheritDoc}
102   * @throws NullPointerException if any task is null
103   */
104  @Override
105  <T extends @Nullable Object> List<Future<T>> invokeAll(
106      Collection<? extends Callable<T>> tasks, long timeout, TimeUnit unit)
107      throws InterruptedException;
108}
