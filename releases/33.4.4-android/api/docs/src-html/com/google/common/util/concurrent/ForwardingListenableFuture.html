015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.GwtCompatible;
018import com.google.common.base.Preconditions;
019import java.util.concurrent.Executor;
020import org.jspecify.annotations.Nullable;
021
022/**
023 * A {@link ListenableFuture} which forwards all its method calls to another future. Subclasses
024 * should override one or more methods to modify the behavior of the backing future as desired per
025 * the <a href="http://en.wikipedia.org/wiki/Decorator_pattern">decorator pattern</a>.
026 *
027 * <p>Most subclasses can just use {@link SimpleForwardingListenableFuture}.
028 *
029 * @author Shardul Deo
030 * @since 4.0
031 */
032@GwtCompatible
033public abstract class ForwardingListenableFuture<V extends @Nullable Object>
034    extends ForwardingFuture<V> implements ListenableFuture<V> {
035
036  /** Constructor for use by subclasses. */
037  protected ForwardingListenableFuture() {}
038
039  @Override
040  protected abstract ListenableFuture<? extends V> delegate();
041
042  @Override
043  public void addListener(Runnable listener, Executor exec) {
044    delegate().addListener(listener, exec);
045  }
046
047  // TODO(cpovirk): Use standard Javadoc form for SimpleForwarding* class and constructor
048  /**
049   * A simplified version of {@link ForwardingListenableFuture} where subclasses can pass in an
050   * already constructed {@link ListenableFuture} as the delegate.
051   *
052   * @since 9.0
053   */
054  public abstract static class SimpleForwardingListenableFuture<V extends @Nullable Object>
055      extends ForwardingListenableFuture<V> {
056    private final ListenableFuture<V> delegate;
057
058    protected SimpleForwardingListenableFuture(ListenableFuture<V> delegate) {
059      this.delegate = Preconditions.checkNotNull(delegate);
060    }
061
062    @Override
063    protected final ListenableFuture<V> delegate() {
064      return delegate;
065    }
066  }
067}
