015package com.google.common.base;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018
019import com.google.common.annotations.GwtIncompatible;
020import com.google.common.annotations.J2ktIncompatible;
021import org.jspecify.annotations.Nullable;
022
023/**
024 * This class provides default values for all Java types, as defined by the JLS.
025 *
026 * @author Ben Yu
027 * @since 1.0
028 */
029@J2ktIncompatible
030@GwtIncompatible
031public final class Defaults {
032  private Defaults() {}
033
034  private static final Double DOUBLE_DEFAULT = 0d;
035  private static final Float FLOAT_DEFAULT = 0f;
036
037  /**
038   * Returns the default value of {@code type} as defined by JLS --- {@code 0} for numbers, {@code
039   * false} for {@code boolean} and {@code '\0'} for {@code char}. For non-primitive types and
040   * {@code void}, {@code null} is returned.
041   */
042  @SuppressWarnings("unchecked")
043  public static <T> @Nullable T defaultValue(Class<T> type) {
044    checkNotNull(type);
045    if (type.isPrimitive()) {
046      if (type == boolean.class) {
047        return (T) Boolean.FALSE;
048      } else if (type == char.class) {
049        return (T) Character.valueOf('\0');
050      } else if (type == byte.class) {
051        return (T) Byte.valueOf((byte) 0);
052      } else if (type == short.class) {
053        return (T) Short.valueOf((short) 0);
054      } else if (type == int.class) {
055        return (T) Integer.valueOf(0);
056      } else if (type == long.class) {
057        return (T) Long.valueOf(0L);
058      } else if (type == float.class) {
059        return (T) FLOAT_DEFAULT;
060      } else if (type == double.class) {
061        return (T) DOUBLE_DEFAULT;
062      }
063    }
064    return null;
065  }
066}
