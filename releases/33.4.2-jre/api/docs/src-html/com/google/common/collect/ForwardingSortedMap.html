017package com.google.common.collect;
018
019import static com.google.common.base.Preconditions.checkArgument;
020
021import com.google.common.annotations.GwtCompatible;
022import java.util.Comparator;
023import java.util.NoSuchElementException;
024import java.util.SortedMap;
025import javax.annotation.CheckForNull;
026import org.checkerframework.checker.nullness.qual.Nullable;
027
028/**
029 * A sorted map which forwards all its method calls to another sorted map. Subclasses should
030 * override one or more methods to modify the behavior of the backing sorted map as desired per the
031 * <a href="http://en.wikipedia.org/wiki/Decorator_pattern">decorator pattern</a>.
032 *
033 * <p><b>Warning:</b> The methods of {@code ForwardingSortedMap} forward <i>indiscriminately</i> to
034 * the methods of the delegate. For example, overriding {@link #put} alone <i>will not</i> change
035 * the behavior of {@link #putAll}, which can lead to unexpected behavior. In this case, you should
036 * override {@code putAll} as well, either providing your own implementation, or delegating to the
037 * provided {@code standardPutAll} method.
038 *
039 * <p><b>{@code default} method warning:</b> This class does <i>not</i> forward calls to {@code
040 * default} methods. Instead, it inherits their default implementations. When those implementations
041 * invoke methods, they invoke methods on the {@code ForwardingSortedMap}.
042 *
043 * <p>Each of the {@code standard} methods, where appropriate, use the comparator of the map to test
044 * equality for both keys and values, unlike {@code ForwardingMap}.
045 *
046 * <p>The {@code standard} methods and the collection views they return are not guaranteed to be
047 * thread-safe, even when all of the methods that they depend on are thread-safe.
048 *
049 * @author Mike Bostock
050 * @author Louis Wasserman
051 * @since 2.0
052 */
053@GwtCompatible
054public abstract class ForwardingSortedMap<K extends @Nullable Object, V extends @Nullable Object>
055    extends ForwardingMap<K, V> implements SortedMap<K, V> {
056  // TODO(lowasser): identify places where thread safety is actually lost
057
058  /** Constructor for use by subclasses. */
059  protected ForwardingSortedMap() {}
060
061  @Override
062  protected abstract SortedMap<K, V> delegate();
063
064  @Override
065  @CheckForNull
066  public Comparator<? super K> comparator() {
067    return delegate().comparator();
068  }
069
070  @Override
071  @ParametricNullness
072  public K firstKey() {
073    return delegate().firstKey();
074  }
075
076  @Override
077  public SortedMap<K, V> headMap(@ParametricNullness K toKey) {
078    return delegate().headMap(toKey);
079  }
080
081  @Override
082  @ParametricNullness
083  public K lastKey() {
084    return delegate().lastKey();
085  }
086
087  @Override
088  public SortedMap<K, V> subMap(@ParametricNullness K fromKey, @ParametricNullness K toKey) {
089    return delegate().subMap(fromKey, toKey);
090  }
091
092  @Override
093  public SortedMap<K, V> tailMap(@ParametricNullness K fromKey) {
094    return delegate().tailMap(fromKey);
095  }
096
097  /**
098   * A sensible implementation of {@link SortedMap#keySet} in terms of the methods of {@code
099   * ForwardingSortedMap}. In many cases, you may wish to override {@link
100   * ForwardingSortedMap#keySet} to forward to this implementation or a subclass thereof.
101   *
102   * @since 15.0
103   */
104  protected class StandardKeySet extends Maps.SortedKeySet<K, V> {
105    /** Constructor for use by subclasses. */
106    public StandardKeySet() {
107      super(ForwardingSortedMap.this);
108    }
109  }
110
111  // unsafe, but worst case is a CCE or NPE is thrown, which callers will be expecting
112  @SuppressWarnings({"unchecked", "nullness"})
113  static int unsafeCompare(
114      @CheckForNull Comparator<?> comparator, @CheckForNull Object o1, @CheckForNull Object o2) {
115    if (comparator == null) {
116      return ((Comparable<@Nullable Object>) o1).compareTo(o2);
117    } else {
118      return ((Comparator<@Nullable Object>) comparator).compare(o1, o2);
119    }
120  }
121
122  /**
123   * A sensible definition of {@link #containsKey} in terms of the {@code firstKey()} method of
124   * {@link #tailMap}. If you override {@link #tailMap}, you may wish to override {@link
125   * #containsKey} to forward to this implementation.
126   *
127   * @since 7.0
128   */
129  @Override
130  protected boolean standardContainsKey(@CheckForNull Object key) {
131    try {
132      // any CCE or NPE will be caught
133      @SuppressWarnings({"unchecked", "nullness"})
134      SortedMap<@Nullable Object, V> self = (SortedMap<@Nullable Object, V>) this;
135      Object ceilingKey = self.tailMap(key).firstKey();
136      return unsafeCompare(comparator(), ceilingKey, key) == 0;
137    } catch (ClassCastException | NoSuchElementException | NullPointerException e) {
138      return false;
139    }
140  }
141
142  /**
143   * A sensible default implementation of {@link #subMap(Object, Object)} in terms of {@link
144   * #headMap(Object)} and {@link #tailMap(Object)}. In some situations, you may wish to override
145   * {@link #subMap(Object, Object)} to forward to this implementation.
146   *
147   * @since 7.0
148   */
149  protected SortedMap<K, V> standardSubMap(K fromKey, K toKey) {
150    checkArgument(unsafeCompare(comparator(), fromKey, toKey) <= 0, "fromKey must be <= toKey");
151    return tailMap(fromKey).headMap(toKey);
152  }
153}
