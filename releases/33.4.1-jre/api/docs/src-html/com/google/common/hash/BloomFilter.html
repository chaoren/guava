044/**
045 * A Bloom filter for instances of {@code T}. A Bloom filter offers an approximate containment test
046 * with one-sided error: if it claims that an element is contained in it, this might be in error,
047 * but if it claims that an element is <i>not</i> contained in it, then this is definitely true.
048 *
049 * <p>If you are unfamiliar with Bloom filters, this nice <a
050 * href="http://llimllib.github.io/bloomfilter-tutorial/">tutorial</a> may help you understand how
051 * they work.
052 *
053 * <p>The false positive probability ({@code FPP}) of a Bloom filter is defined as the probability
054 * that {@linkplain #mightContain(Object)} will erroneously return {@code true} for an object that
055 * has not actually been put in the {@code BloomFilter}.
056 *
057 * <p>Bloom filters are serializable. They also support a more compact serial representation via the
058 * {@link #writeTo} and {@link #readFrom} methods. Both serialized forms will continue to be
059 * supported by future versions of this library. However, serial forms generated by newer versions
060 * of the code may not be readable by older versions of the code (e.g., a serialized Bloom filter
061 * generated today may <i>not</i> be readable by a binary that was compiled 6 months ago).
062 *
063 * <p>As of Guava 23.0, this class is thread-safe and lock-free. It internally uses atomics and
064 * compare-and-swap to ensure correctness when multiple threads are used to access it.
065 *
066 * @param <T> the type of instances that the {@code BloomFilter} accepts
067 * @author Dimitris Andreou
068 * @author Kevin Bourrillion
069 * @since 11.0 (thread-safe since 23.0)
070 */
071@Beta
072public final class BloomFilter<T extends @Nullable Object> implements Predicate<T>, Serializable {
073  /**
074   * A strategy to translate T instances, to {@code numHashFunctions} bit indexes.
075   *
076   * <p>Implementations should be collections of pure functions (i.e. stateless).
077   */
078  interface Strategy extends java.io.Serializable {
079
080    /**
081     * Sets {@code numHashFunctions} bits of the given bit array, by hashing a user element.
082     *
083     * <p>Returns whether any bits changed as a result of this operation.
084     */
085    <T extends @Nullable Object> boolean put(
086        @ParametricNullness T object,
087        Funnel<? super T> funnel,
088        int numHashFunctions,
089        LockFreeBitArray bits);
090
091    /**
092     * Queries {@code numHashFunctions} bits of the given bit array, by hashing a user element;
093     * returns {@code true} if and only if all selected bits are set.
094     */
095    <T extends @Nullable Object> boolean mightContain(
096        @ParametricNullness T object,
097        Funnel<? super T> funnel,
098        int numHashFunctions,
099        LockFreeBitArray bits);
100
101    /**
102     * Identifier used to encode this strategy, when marshalled as part of a BloomFilter. Only
103     * values in the [-128, 127] range are valid for the compact serial form. Non-negative values
104     * are reserved for enums defined in BloomFilterStrategies; negative values are reserved for any
105     * custom, stateful strategy we may define (e.g. any kind of strategy that would depend on user
106     * input).
107     */
108    int ordinal();
109  }
110
111  /** The bit set of the BloomFilter (not necessarily power of 2!) */
112  private final LockFreeBitArray bits;
113
114  /** Number of hashes per element */
115  private final int numHashFunctions;
116
117  /** The funnel to translate Ts to bytes */
118  private final Funnel<? super T> funnel;
119
120  /** The strategy we employ to map an element T to {@code numHashFunctions} bit indexes. */
121  private final Strategy strategy;
122
123  /** Natural logarithm of 2, used to optimize calculations in Bloom filter sizing. */
124  private static final double LOG_TWO = Math.log(2);
125
126  /** Square of the natural logarithm of 2, reused to optimize the bit size calculation. */
127  private static final double SQUARED_LOG_TWO = LOG_TWO * LOG_TWO;

129  /** Creates a BloomFilter. */
130  private BloomFilter(
131      LockFreeBitArray bits, int numHashFunctions, Funnel<? super T> funnel, Strategy strategy) {
132    checkArgument(numHashFunctions > 0, "numHashFunctions (%s) must be > 0", numHashFunctions);
133    checkArgument(
134        numHashFunctions <= 255, "numHashFunctions (%s) must be <= 255", numHashFunctions);
135    this.bits = checkNotNull(bits);
136    this.numHashFunctions = numHashFunctions;
137    this.funnel = checkNotNull(funnel);
138    this.strategy = checkNotNull(strategy);
139  }

169  /**
170   * Puts an element into this {@code BloomFilter}. Ensures that subsequent invocations of {@link
171   * #mightContain(Object)} with the same element will always return {@code true}.
172   *
173   * @return true if the Bloom filter's bits changed as a result of this operation. If the bits
174   *     changed, this is <i>definitely</i> the first time {@code object} has been added to the
175   *     filter. If the bits haven't changed, this <i>might</i> be the first time {@code object} has
176   *     been added to the filter. Note that {@code put(t)} always returns the <i>opposite</i>
177   *     result to what {@code mightContain(t)} would have returned at the time it is called.
178   * @since 12.0 (present in 11.0 with {@code void} return type})
179   */
180  @CanIgnoreReturnValue
181  public boolean put(@ParametricNullness T object) {
182    return strategy.put(object, funnel, numHashFunctions, bits);
183  }

185  /**
186   * Returns the probability that {@linkplain #mightContain(Object)} will erroneously return {@code
187   * true} for an object that has not actually been put in the {@code BloomFilter}.
188   *
189   * <p>Ideally, this number should be close to the {@code fpp} parameter passed in {@linkplain
190   * #create(Funnel, int, double)}, or smaller. If it is significantly higher, it is usually the
191   * case that too many elements (more than expected) have been put in the {@code BloomFilter},
192   * degenerating it.
193   *
194   * @since 14.0 (since 11.0 as expectedFalsePositiveProbability())
195   */
196  public double expectedFpp() {
197    return Math.pow((double) bits.bitCount() / bitSize(), numHashFunctions);
198  }
199
200  /**
201   * Returns an estimate for the total number of distinct elements that have been added to this
202   * Bloom filter. This approximation is reasonably accurate if it does not exceed the value of
203   * {@code expectedInsertions} that was used when constructing the filter.
204   *
205   * @since 22.0
206   */
207  public long approximateElementCount() {
208    long bitSize = bits.bitSize();
209    long bitCount = bits.bitCount();
210
211    /**
212     * Each insertion is expected to reduce the # of clear bits by a factor of
213     * `numHashFunctions/bitSize`. So, after n insertions, expected bitCount is `bitSize * (1 - (1 -
214     * numHashFunctions/bitSize)^n)`. Solving that for n, and approximating `ln x` as `x - 1` when x
215     * is close to 1 (why?), gives the following formula.
216     */
217    double fractionOfBitsSet = (double) bitCount / bitSize;
218    return DoubleMath.roundToLong(
219        -Math.log1p(-fractionOfBitsSet) * bitSize / numHashFunctions, RoundingMode.HALF_UP);
220  }

222  /** Returns the number of bits in the underlying bit array. */
223  @VisibleForTesting

225    return bits.bitSize();

228  /**
229   * Determines whether a given Bloom filter is compatible with this Bloom filter. For two Bloom
230   * filters to be compatible, they must:
231   *
232   * <ul>
233   *   <li>not be the same instance
234   *   <li>have the same number of hash functions
235   *   <li>have the same bit size
236   *   <li>have the same strategy
237   *   <li>have equal funnels
238   * </ul>
239   *
240   * @param that The Bloom filter to check for compatibility.
241   * @since 15.0
242   */
243  public boolean isCompatible(BloomFilter<T> that) {
244    checkNotNull(that);
245    return this != that
246        && this.numHashFunctions == that.numHashFunctions
247        && this.bitSize() == that.bitSize()
248        && this.strategy.equals(that.strategy)
249        && this.funnel.equals(that.funnel);
250  }
251
252  /**
253   * Combines this Bloom filter with another Bloom filter by performing a bitwise OR of the
254   * underlying data. The mutations happen to <b>this</b> instance. Callers must ensure the Bloom
255   * filters are appropriately sized to avoid saturating them.
256   *
257   * @param that The Bloom filter to combine this Bloom filter with. It is not mutated.
258   * @throws IllegalArgumentException if {@code isCompatible(that) == false}
259   * @since 15.0
260   */
261  public void putAll(BloomFilter<T> that) {
262    checkNotNull(that);
263    checkArgument(this != that, "Cannot combine a BloomFilter with itself.");
264    checkArgument(
265        this.numHashFunctions == that.numHashFunctions,
266        "BloomFilters must have the same number of hash functions (%s != %s)",
267        this.numHashFunctions,
268        that.numHashFunctions);
269    checkArgument(
270        this.bitSize() == that.bitSize(),
271        "BloomFilters must have the same size underlying bit arrays (%s != %s)",
272        this.bitSize(),
273        that.bitSize());
274    checkArgument(
275        this.strategy.equals(that.strategy),
276        "BloomFilters must have equal strategies (%s != %s)",
277        this.strategy,
278        that.strategy);
279    checkArgument(
280        this.funnel.equals(that.funnel),
281        "BloomFilters must have equal funnels (%s != %s)",
282        this.funnel,
283        that.funnel);
284    this.bits.putAll(that.bits);
285  }

287  @Override
288  public boolean equals(@CheckForNull Object object) {
289    if (object == this) {
290      return true;
291    }
292    if (object instanceof BloomFilter) {
293      BloomFilter<?> that = (BloomFilter<?>) object;
294      return this.numHashFunctions == that.numHashFunctions
295          && this.funnel.equals(that.funnel)
296          && this.bits.equals(that.bits)
297          && this.strategy.equals(that.strategy);
298    }
299    return false;
300  }

333  /**
334   * Returns a {@code Collector} expecting the specified number of insertions, and yielding a {@link
335   * BloomFilter} with the specified expected false positive probability.
336   *
337   * <p>Note that if the {@code Collector} receives significantly more elements than specified, the
338   * resulting {@code BloomFilter} will suffer a sharp deterioration of its false positive
339   * probability.
340   *
341   * <p>The constructed {@code BloomFilter} will be serializable if the provided {@code Funnel<T>}
342   * is.
343   *
344   * <p>It is recommended that the funnel be implemented as a Java enum. This has the benefit of
345   * ensuring proper serialization and deserialization, which is important since {@link #equals}
346   * also relies on object identity of funnels.
347   *
348   * @param funnel the funnel of T's that the constructed {@code BloomFilter} will use
349   * @param expectedInsertions the number of expected insertions to the constructed {@code
350   *     BloomFilter}; must be positive
351   * @param fpp the desired false positive probability (must be positive and less than 1.0)
352   * @return a {@code Collector} generating a {@code BloomFilter} of the received elements
353   * @since 23.0 (but only since 33.4.0 in the Android flavor)
354   */
355  public static <T extends @Nullable Object> Collector<T, ?, BloomFilter<T>> toBloomFilter(
356      Funnel<? super T> funnel, long expectedInsertions, double fpp) {
357    checkNotNull(funnel);
358    checkArgument(
359        expectedInsertions >= 0, "Expected insertions (%s) must be >= 0", expectedInsertions);
360    checkArgument(fpp > 0.0, "False positive probability (%s) must be > 0.0", fpp);
361    checkArgument(fpp < 1.0, "False positive probability (%s) must be < 1.0", fpp);
362    return Collector.of(
363        () -> BloomFilter.create(funnel, expectedInsertions, fpp),
364        BloomFilter::put,
365        (bf1, bf2) -> {
366          bf1.putAll(bf2);
367          return bf1;
368        },
369        Collector.Characteristics.UNORDERED,
370        Collector.Characteristics.CONCURRENT);
371  }

424  @VisibleForTesting
425  static <T extends @Nullable Object> BloomFilter<T> create(
426      Funnel<? super T> funnel, long expectedInsertions, double fpp, Strategy strategy) {
427    checkNotNull(funnel);
428    checkArgument(
429        expectedInsertions >= 0, "Expected insertions (%s) must be >= 0", expectedInsertions);
430    checkArgument(fpp > 0.0, "False positive probability (%s) must be > 0.0", fpp);
431    checkArgument(fpp < 1.0, "False positive probability (%s) must be < 1.0", fpp);
432    checkNotNull(strategy);
433
434    if (expectedInsertions == 0) {
435      expectedInsertions = 1;
436    }
437    /*
438     * TODO(user): Put a warning in the javadoc about tiny fpp values, since the resulting size
439     * is proportional to -log(p), but there is not much of a point after all, e.g.
440     * optimalM(1000, 0.0000000000000001) = 76680 which is less than 10kb. Who cares!
441     */
442    long numBits = optimalNumOfBits(expectedInsertions, fpp);
443    int numHashFunctions = optimalNumOfHashFunctions(fpp);
444    try {
445      return new BloomFilter<>(new LockFreeBitArray(numBits), numHashFunctions, funnel, strategy);
446    } catch (IllegalArgumentException e) {
447      throw new IllegalArgumentException("Could not create BloomFilter of " + numBits + " bits", e);
448    }
449  }

500  // Cheat sheet:
501  //
502  // m: total bits
503  // n: expected insertions
504  // b: m/n, bits per insertion
505  // p: expected false positive probability
506  //
507  // 1) Optimal k = b * ln2
508  // 2) p = (1 - e ^ (-kn/m))^k
509  // 3) For optimal k: p = 2 ^ (-k) ~= 0.6185^b
510  // 4) For optimal k: m = -nlnp / ((ln2) ^ 2)
511
512  /**
513   * Computes the optimal number of hash functions (k) for a given false positive probability (p).
514   *
515   * <p>See http://en.wikipedia.org/wiki/File:Bloom_filter_fp_probability.svg for the formula.
516   *
517   * @param p desired false positive probability (must be between 0 and 1, exclusive)
518   */
519  @VisibleForTesting
520  static int optimalNumOfHashFunctions(double p) {
521    // -log(p) / log(2), ensuring the result is rounded to avoid truncation.
522    return max(1, (int) Math.round(-Math.log(p) / LOG_TWO));
523  }
524
525  /**
526   * Computes m (total bits of Bloom filter) which is expected to achieve, for the specified
527   * expected insertions, the required false positive probability.
528   *
529   * <p>See http://en.wikipedia.org/wiki/Bloom_filter#Probability_of_false_positives for the
530   * formula.
531   *
532   * @param n expected insertions (must be positive)
533   * @param p false positive rate (must be 0 < p < 1)
534   */
535  @VisibleForTesting
536  static long optimalNumOfBits(long n, double p) {
537    if (p == 0) {
538      p = Double.MIN_VALUE;
539    }
540    return (long) (-n * Math.log(p) / SQUARED_LOG_TWO);
541  }

551  private static class SerialForm<T extends @Nullable Object> implements Serializable {
552    final long[] data;
553    final int numHashFunctions;
554    final Funnel<? super T> funnel;
555    final Strategy strategy;
556
557    SerialForm(BloomFilter<T> bf) {
558      this.data = LockFreeBitArray.toPlainArray(bf.bits.data);
559      this.numHashFunctions = bf.numHashFunctions;
560      this.funnel = bf.funnel;
561      this.strategy = bf.strategy;
562    }
563
564    Object readResolve() {
565      return new BloomFilter<T>(new LockFreeBitArray(data), numHashFunctions, funnel, strategy);
566    }
567
568    private static final long serialVersionUID = 1;
569  }

571  /**
572   * Writes this {@code BloomFilter} to an output stream, with a custom format (not Java
573   * serialization). This has been measured to save at least 400 bytes compared to regular
574   * serialization.
575   *
576   * <p>Use {@linkplain #readFrom(InputStream, Funnel)} to reconstruct the written BloomFilter.
577   */
578  public void writeTo(OutputStream out) throws IOException {
579    // Serial form:
580    // 1 signed byte for the strategy
581    // 1 unsigned byte for the number of hash functions
582    // 1 big endian int, the number of longs in our bitset
583    // N big endian longs of our bitset
584    DataOutputStream dout = new DataOutputStream(out);
585    dout.writeByte(SignedBytes.checkedCast(strategy.ordinal()));
586    dout.writeByte(UnsignedBytes.checkedCast(numHashFunctions)); // note: checked at the c'tor
587    dout.writeInt(bits.data.length());
588    for (int i = 0; i < bits.data.length(); i++) {
589      dout.writeLong(bits.data.get(i));
590    }
591  }

593  /**
594   * Reads a byte stream, which was written by {@linkplain #writeTo(OutputStream)}, into a {@code
595   * BloomFilter}.
596   *
597   * <p>The {@code Funnel} to be used is not encoded in the stream, so it must be provided here.
598   * <b>Warning:</b> the funnel provided <b>must</b> behave identically to the one used to populate
599   * the original Bloom filter!
600   *
601   * @throws IOException if the InputStream throws an {@code IOException}, or if its data does not
602   *     appear to be a BloomFilter serialized using the {@linkplain #writeTo(OutputStream)} method.
603   */
604  @SuppressWarnings("CatchingUnchecked") // sneaky checked exception
605  public static <T extends @Nullable Object> BloomFilter<T> readFrom(
606      InputStream in, Funnel<? super T> funnel) throws IOException {
607    checkNotNull(in, "InputStream");
608    checkNotNull(funnel, "Funnel");
609    int strategyOrdinal = -1;
610    int numHashFunctions = -1;
611    int dataLength = -1;
612    try {
613      DataInputStream din = new DataInputStream(in);
614      // currently this assumes there is no negative ordinal; will have to be updated if we
615      // add non-stateless strategies (for which we've reserved negative ordinals; see
616      // Strategy.ordinal()).
617      strategyOrdinal = din.readByte();
618      numHashFunctions = UnsignedBytes.toInt(din.readByte());
619      dataLength = din.readInt();
620
621      Strategy strategy = BloomFilterStrategies.values()[strategyOrdinal];
622
623      LockFreeBitArray dataArray = new LockFreeBitArray(LongMath.checkedMultiply(dataLength, 64L));
624      for (int i = 0; i < dataLength; i++) {
625        dataArray.putData(i, din.readLong());
626      }
627
628      return new BloomFilter<>(dataArray, numHashFunctions, funnel, strategy);
629    } catch (IOException e) {
630      throw e;
631    } catch (Exception e) { // sneaky checked exception
632      String message =
633          "Unable to deserialize BloomFilter from InputStream."
634              + " strategyOrdinal: "
635              + strategyOrdinal
636              + " numHashFunctions: "
637              + numHashFunctions
638              + " dataLength: "
639              + dataLength;
640      throw new IOException(message, e);
641    }
642  }
