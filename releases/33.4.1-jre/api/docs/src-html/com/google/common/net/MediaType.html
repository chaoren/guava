048/**
049 * Represents an <a href="http://en.wikipedia.org/wiki/Internet_media_type">Internet Media Type</a>
050 * (also known as a MIME Type or Content Type). This class also supports the concept of media ranges
051 * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1">defined by HTTP/1.1</a>.
052 * As such, the {@code *} character is treated as a wildcard and is used to represent any acceptable
053 * type or subtype value. A media type may not have wildcard type with a declared subtype. The
054 * {@code *} character has no special meaning as part of a parameter. All values for type, subtype,
055 * parameter attributes or parameter values must be valid according to RFCs <a
056 * href="https://tools.ietf.org/html/rfc2045">2045</a> and <a
057 * href="https://tools.ietf.org/html/rfc2046">2046</a>.
058 *
059 * <p>All portions of the media type that are case-insensitive (type, subtype, parameter attributes)
060 * are normalized to lowercase. The value of the {@code charset} parameter is normalized to
061 * lowercase, but all others are left as-is.
062 *
063 * <p>Note that this specifically does <strong>not</strong> represent the value of the MIME {@code
064 * Content-Type} header and as such has no support for header-specific considerations such as line
065 * folding and comments.
066 *
067 * <p>For media types that take a charset the predefined constants default to UTF-8 and have a
068 * "_UTF_8" suffix. To get a version without a character set, use {@link #withoutParameters}.
069 *
070 * @since 12.0
071 * @author Gregory Kick
072 */
073@GwtCompatible
074@Immutable
075public final class MediaType {
076  private static final String CHARSET_ATTRIBUTE = "charset";
077  private static final ImmutableListMultimap<String, String> UTF_8_CONSTANT_PARAMETERS =
078      ImmutableListMultimap.of(CHARSET_ATTRIBUTE, Ascii.toLowerCase(UTF_8.name()));

080  /** Matcher for type, subtype and attributes. */
081  private static final CharMatcher TOKEN_MATCHER =
082      ascii()
083          .and(javaIsoControl().negate())
084          .and(CharMatcher.isNot(' '))
085          .and(CharMatcher.noneOf("()<>@,;:\\\"/[]?="));

105  private static final Map<MediaType, MediaType> KNOWN_TYPES = Maps.newHashMap();
106
107  private static MediaType createConstant(String type, String subtype) {
108    MediaType mediaType =
109        addKnownType(new MediaType(type, subtype, ImmutableListMultimap.<String, String>of()));
110    mediaType.parsedCharset = Optional.absent();
111    return mediaType;
112  }
113
114  private static MediaType createConstantUtf8(String type, String subtype) {
115    MediaType mediaType = addKnownType(new MediaType(type, subtype, UTF_8_CONSTANT_PARAMETERS));
116    mediaType.parsedCharset = Optional.of(UTF_8);
117    return mediaType;
118  }
119
120  private static MediaType addKnownType(MediaType mediaType) {
121    KNOWN_TYPES.put(mediaType, mediaType);
122    return mediaType;
123  }

837  /**
838   * Returns an optional charset for the value of the charset parameter if it is specified.
839   *
840   * @throws IllegalStateException if multiple charset values have been set for this media type
841   * @throws IllegalCharsetNameException if a charset value is present, but illegal
842   * @throws UnsupportedCharsetException if a charset value is present, but no support is available
843   *     in this instance of the Java virtual machine
844   */
845  public Optional<Charset> charset() {
846    // racy single-check idiom, this is safe because Optional is immutable.
847    Optional<Charset> local = parsedCharset;
848    if (local == null) {
849      String value = null;
850      local = Optional.absent();
851      for (String currentValue : parameters.get(CHARSET_ATTRIBUTE)) {
852        if (value == null) {
853          value = currentValue;
854          local = Optional.of(Charset.forName(value));
855        } else if (!value.equals(currentValue)) {
856          throw new IllegalStateException(
857              "Multiple charset values defined: " + value + ", " + currentValue);
858        }
859      }
860      parsedCharset = local;
861    }
862    return local;
863  }

882  /**
883   * <em>Replaces</em> all parameters with the given attribute with parameters using the given
884   * values. If there are no values, any existing parameters with the given attribute are removed.
885   *
886   * @throws IllegalArgumentException if either {@code attribute} or {@code values} is invalid
887   * @since 24.0
888   */
889  public MediaType withParameters(String attribute, Iterable<String> values) {
890    checkNotNull(attribute);
891    checkNotNull(values);
892    String normalizedAttribute = normalizeToken(attribute);
893    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
894    for (Entry<String, String> entry : parameters.entries()) {
895      String key = entry.getKey();
896      if (!normalizedAttribute.equals(key)) {
897        builder.put(key, entry.getValue());
898      }
899    }
900    for (String value : values) {
901      builder.put(normalizedAttribute, normalizeParameterValue(normalizedAttribute, value));
902    }
903    MediaType mediaType = new MediaType(type, subtype, builder.build());
904    // if the attribute isn't charset, we can just inherit the current parsedCharset
905    if (!normalizedAttribute.equals(CHARSET_ATTRIBUTE)) {
906      mediaType.parsedCharset = this.parsedCharset;
907    }
908    // Return one of the constants if the media type is a known type.
909    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
910  }

914   * given value. If multiple parameters with the same attributes are necessary use {@link
915   * #withParameters(String, Iterable)}. Prefer {@link #withCharset} for setting the {@code charset}
916   * parameter when using a {@link Charset} object.
917   *
918   * @throws IllegalArgumentException if either {@code attribute} or {@code value} is invalid
919   */
920  public MediaType withParameter(String attribute, String value) {

926   * charset} parameter set to the {@link Charset#name name} of the given charset. Only one {@code
927   * charset} parameter will be present on the new instance regardless of the number set on this

946  /**
947   * Returns {@code true} if this instance falls within the range (as defined by <a
948   * href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html">the HTTP Accept header</a>) given
949   * by the argument according to three criteria:
950   *
951   * <ol>
952   *   <li>The type of the argument is the wildcard or equal to the type of this instance.
953   *   <li>The subtype of the argument is the wildcard or equal to the subtype of this instance.
954   *   <li>All of the parameters present in the argument are present in this instance.
955   * </ol>
956   *
957   * <p>For example:
958   *
959   * <pre>{@code
960   * PLAIN_TEXT_UTF_8.is(PLAIN_TEXT_UTF_8) // true
961   * PLAIN_TEXT_UTF_8.is(HTML_UTF_8) // false
962   * PLAIN_TEXT_UTF_8.is(ANY_TYPE) // true
963   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE) // true
964   * PLAIN_TEXT_UTF_8.is(ANY_IMAGE_TYPE) // false
965   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_8)) // true
966   * PLAIN_TEXT_UTF_8.withoutParameters().is(ANY_TEXT_TYPE.withCharset(UTF_8)) // false
967   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_16)) // false
968   * }</pre>
969   *
970   * <p>Note that while it is possible to have the same parameter declared multiple times within a
971   * media type this method does not consider the number of occurrences of a parameter. For example,
972   * {@code "text/plain; charset=UTF-8"} satisfies {@code "text/plain; charset=UTF-8;
973   * charset=UTF-8"}.
974   */
975  public boolean is(MediaType mediaTypeRange) {
976    return (mediaTypeRange.type.equals(WILDCARD) || mediaTypeRange.type.equals(this.type))
977        && (mediaTypeRange.subtype.equals(WILDCARD) || mediaTypeRange.subtype.equals(this.subtype))
978        && this.parameters.entries().containsAll(mediaTypeRange.parameters.entries());
979  }

981  /**
982   * Creates a new media type with the given type and subtype.
983   *
984   * @throws IllegalArgumentException if type or subtype is invalid or if a wildcard is used for the
985   *     type, but not the subtype.
986   */
987  public static MediaType create(String type, String subtype) {
988    MediaType mediaType = create(type, subtype, ImmutableListMultimap.<String, String>of());
989    mediaType.parsedCharset = Optional.absent();
990    return mediaType;
991  }

993  private static MediaType create(
994      String type, String subtype, Multimap<String, String> parameters) {
995    checkNotNull(type);
996    checkNotNull(subtype);
997    checkNotNull(parameters);
998    String normalizedType = normalizeToken(type);
999    String normalizedSubtype = normalizeToken(subtype);
1000    checkArgument(
1001        !WILDCARD.equals(normalizedType) || WILDCARD.equals(normalizedSubtype),
1002        "A wildcard type cannot be used with a non-wildcard subtype");
1003    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
1004    for (Entry<String, String> entry : parameters.entries()) {
1005      String attribute = normalizeToken(entry.getKey());
1006      builder.put(attribute, normalizeParameterValue(attribute, entry.getValue()));
1007    }
1008    MediaType mediaType = new MediaType(normalizedType, normalizedSubtype, builder.build());
1009    // Return one of the constants if the media type is a known type.
1010    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
1011  }

1067  private static String normalizeToken(String token) {
1068    checkArgument(TOKEN_MATCHER.matchesAllOf(token));
1069    checkArgument(!token.isEmpty());
1070    return Ascii.toLowerCase(token);
1071  }

1073  private static String normalizeParameterValue(String attribute, String value) {
1074    checkNotNull(value); // for GWT
1075    checkArgument(ascii().matchesAllOf(value), "parameter values must be ASCII: %s", value);
1076    return CHARSET_ATTRIBUTE.equals(attribute) ? Ascii.toLowerCase(value) : value;
1077  }

1079  /**
1080   * Parses a media type from its string representation.
1081   *
1082   * @throws IllegalArgumentException if the input is not parsable
1083   */
1084  @CanIgnoreReturnValue // TODO(b/219820829): consider removing
1085  public static MediaType parse(String input) {
1086    checkNotNull(input);
1087    Tokenizer tokenizer = new Tokenizer(input);
1088    try {
1089      String type = tokenizer.consumeToken(TOKEN_MATCHER);
1090      consumeSeparator(tokenizer, '/');
1091      String subtype = tokenizer.consumeToken(TOKEN_MATCHER);
1092      ImmutableListMultimap.Builder<String, String> parameters = ImmutableListMultimap.builder();
1093      while (tokenizer.hasMore()) {
1094        consumeSeparator(tokenizer, ';');
1095        String attribute = tokenizer.consumeToken(TOKEN_MATCHER);
1096        consumeSeparator(tokenizer, '=');
1097        String value;
1098        if ('"' == tokenizer.previewChar()) {
1099          tokenizer.consumeCharacter('"');
1100          StringBuilder valueBuilder = new StringBuilder();
1101          while ('"' != tokenizer.previewChar()) {
1102            if ('\\' == tokenizer.previewChar()) {
1103              tokenizer.consumeCharacter('\\');
1104              valueBuilder.append(tokenizer.consumeCharacter(ascii()));
1105            } else {
1106              valueBuilder.append(tokenizer.consumeToken(QUOTED_TEXT_MATCHER));
1107            }
1108          }
1109          value = valueBuilder.toString();
1110          tokenizer.consumeCharacter('"');
1111        } else {
1112          value = tokenizer.consumeToken(TOKEN_MATCHER);
1113        }
1114        parameters.put(attribute, value);
1115      }
1116      return create(type, subtype, parameters.build());
1117    } catch (IllegalStateException e) {
1118      throw new IllegalArgumentException("Could not parse '" + input + "'", e);
1119    }
1120  }

1122  private static void consumeSeparator(Tokenizer tokenizer, char c) {
1123    tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
1124    tokenizer.consumeCharacter(c);
1125    tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
1126  }

1128  private static final class Tokenizer {
1129    final String input;
1130    int position = 0;
1131
1132    Tokenizer(String input) {
1133      this.input = input;
1134    }
1135
1136    @CanIgnoreReturnValue
1137    String consumeTokenIfPresent(CharMatcher matcher) {
1138      checkState(hasMore());
1139      int startPosition = position;
1140      position = matcher.negate().indexIn(input, startPosition);
1141      return hasMore() ? input.substring(startPosition, position) : input.substring(startPosition);
1142    }
1143
1144    String consumeToken(CharMatcher matcher) {
1145      int startPosition = position;
1146      String token = consumeTokenIfPresent(matcher);
1147      checkState(position != startPosition);
1148      return token;
1149    }
1150
1151    char consumeCharacter(CharMatcher matcher) {
1152      checkState(hasMore());
1153      char c = previewChar();
1154      checkState(matcher.matches(c));
1155      position++;
1156      return c;
1157    }
1158
1159    @CanIgnoreReturnValue
1160    char consumeCharacter(char c) {
1161      checkState(hasMore());
1162      checkState(previewChar() == c);
1163      position++;
1164      return c;
1165    }
1166
1167    char previewChar() {
1168      checkState(hasMore());
1169      return input.charAt(position);
1170    }
1171
1172    boolean hasMore() {
1173      return (position >= 0) && (position < input.length());
1174    }
1175  }

1203  private static final MapJoiner PARAMETER_JOINER = Joiner.on("; ").withKeyValueSeparator("=");
1204
1205  /**
1206   * Returns the string representation of this media type in the format described in <a
1207   * href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>.
1208   */
1209  @Override
1210  public String toString() {
1211    // racy single-check idiom, safe because String is immutable
1212    String result = toString;
1213    if (result == null) {
1214      result = computeToString();
1215      toString = result;
1216    }
1217    return result;
1218  }
1219
1220  private String computeToString() {
1221    StringBuilder builder = new StringBuilder().append(type).append('/').append(subtype);
1222    if (!parameters.isEmpty()) {
1223      builder.append("; ");
1224      Multimap<String, String> quotedParameters =
1225          Multimaps.transformValues(
1226              parameters,
1227              (String value) ->
1228                  (TOKEN_MATCHER.matchesAllOf(value) && !value.isEmpty())
1229                      ? value
1230                      : escapeAndQuote(value));
1231      PARAMETER_JOINER.appendTo(builder, quotedParameters.entries());
1232    }
1233    return builder.toString();
1234  }
1235
1236  private static String escapeAndQuote(String value) {
1237    StringBuilder escaped = new StringBuilder(value.length() + 16).append('"');
1238    for (int i = 0; i < value.length(); i++) {
1239      char ch = value.charAt(i);
1240      if (ch == '\r' || ch == '\\' || ch == '"') {
1241        escaped.append('\\');
1242      }
1243      escaped.append(ch);
1244    }
1245    return escaped.append('"').toString();
1246  }
