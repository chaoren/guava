017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkNotNull;
020
021import com.google.common.annotations.Beta;
022import com.google.common.base.Function;
023import com.google.common.collect.ImmutableMap;
024import com.google.common.collect.Maps;
025import com.google.errorprone.annotations.CanIgnoreReturnValue;
026import com.google.errorprone.annotations.Immutable;
027import java.util.Map;
028
029/**
030 * A {@link Network} whose elements and structural relationships will never change. Instances of
031 * this class may be obtained with {@link #copyOf(Network)}.
032 *
033 * <p>See the Guava User's Guide's <a
034 * href="https://github.com/google/guava/wiki/GraphsExplained#immutable-implementations">discussion
035 * of the {@code Immutable*} types</a> for more information on the properties and guarantees
036 * provided by this class.
037 *
038 * @author James Sexton
039 * @author Joshua O'Madadhain
040 * @author Omar Darwish
041 * @author Jens Nyman
042 * @param <N> Node parameter type
043 * @param <E> Edge parameter type
044 * @since 20.0
045 */
046@Beta
047@Immutable(containerOf = {"N", "E"})
048@SuppressWarnings("Immutable") // Extends StandardNetwork but uses ImmutableMaps.
049public final class ImmutableNetwork<N, E> extends StandardNetwork<N, E> {
050
051  private ImmutableNetwork(Network<N, E> network) {
052    super(
053        NetworkBuilder.from(network), getNodeConnections(network), getEdgeToReferenceNode(network));
054  }
055
056  /** Returns an immutable copy of {@code network}. */
057  public static <N, E> ImmutableNetwork<N, E> copyOf(Network<N, E> network) {
058    return (network instanceof ImmutableNetwork)
059        ? (ImmutableNetwork<N, E>) network
060        : new ImmutableNetwork<N, E>(network);
061  }
062
063  /**
064   * Simply returns its argument.
065   *
066   * @deprecated no need to use this
067   */
068  @Deprecated
069  public static <N, E> ImmutableNetwork<N, E> copyOf(ImmutableNetwork<N, E> network) {
070    return checkNotNull(network);
071  }
072
073  @Override
074  public ImmutableGraph<N> asGraph() {
075    return new ImmutableGraph<>(super.asGraph()); // safe because the view is effectively immutable
076  }
077
078  private static <N, E> Map<N, NetworkConnections<N, E>> getNodeConnections(Network<N, E> network) {
079    // ImmutableMap.Builder maintains the order of the elements as inserted, so the map will have
080    // whatever ordering the network's nodes do, so ImmutableSortedMap is unnecessary even if the
081    // input nodes are sorted.
082    ImmutableMap.Builder<N, NetworkConnections<N, E>> nodeConnections = ImmutableMap.builder();
083    for (N node : network.nodes()) {
084      nodeConnections.put(node, connectionsOf(network, node));
085    }
086    return nodeConnections.buildOrThrow();
087  }
088
089  private static <N, E> Map<E, N> getEdgeToReferenceNode(Network<N, E> network) {
090    // ImmutableMap.Builder maintains the order of the elements as inserted, so the map will have
091    // whatever ordering the network's edges do, so ImmutableSortedMap is unnecessary even if the
092    // input edges are sorted.
093    ImmutableMap.Builder<E, N> edgeToReferenceNode = ImmutableMap.builder();
094    for (E edge : network.edges()) {
095      edgeToReferenceNode.put(edge, network.incidentNodes(edge).nodeU());
096    }
097    return edgeToReferenceNode.buildOrThrow();
098  }
099
100  private static <N, E> NetworkConnections<N, E> connectionsOf(Network<N, E> network, N node) {
101    if (network.isDirected()) {
102      Map<E, N> inEdgeMap = Maps.asMap(network.inEdges(node), sourceNodeFn(network));
103      Map<E, N> outEdgeMap = Maps.asMap(network.outEdges(node), targetNodeFn(network));
104      int selfLoopCount = network.edgesConnecting(node, node).size();
105      return network.allowsParallelEdges()
106          ? DirectedMultiNetworkConnections.ofImmutable(inEdgeMap, outEdgeMap, selfLoopCount)
107          : DirectedNetworkConnections.ofImmutable(inEdgeMap, outEdgeMap, selfLoopCount);
108    } else {
109      Map<E, N> incidentEdgeMap =
110          Maps.asMap(network.incidentEdges(node), adjacentNodeFn(network, node));
111      return network.allowsParallelEdges()
112          ? UndirectedMultiNetworkConnections.ofImmutable(incidentEdgeMap)
113          : UndirectedNetworkConnections.ofImmutable(incidentEdgeMap);
114    }
115  }
116
117  private static <N, E> Function<E, N> sourceNodeFn(Network<N, E> network) {
118    return (E edge) -> network.incidentNodes(edge).source();
119  }
120
121  private static <N, E> Function<E, N> targetNodeFn(Network<N, E> network) {
122    return (E edge) -> network.incidentNodes(edge).target();
123  }
124
125  private static <N, E> Function<E, N> adjacentNodeFn(Network<N, E> network, N node) {
126    return (E edge) -> network.incidentNodes(edge).adjacentNode(node);
127  }
128
129  /**
130   * A builder for creating {@link ImmutableNetwork} instances, especially {@code static final}
131   * networks. Example:
132   *
133   * <pre>{@code
134   * static final ImmutableNetwork<City, Train> TRAIN_NETWORK =
135   *     NetworkBuilder.undirected()
136   *         .allowsParallelEdges(true)
137   *         .<City, Train>immutable()
138   *         .addEdge(PARIS, BRUSSELS, Thalys.trainNumber("1111"))
139   *         .addEdge(PARIS, BRUSSELS, RegionalTrain.trainNumber("2222"))
140   *         .addEdge(LONDON, PARIS, Eurostar.trainNumber("3333"))
141   *         .addEdge(LONDON, BRUSSELS, Eurostar.trainNumber("4444"))
142   *         .addNode(REYKJAVIK)
143   *         .build();
144   * }</pre>
145   *
146   * <p>Builder instances can be reused; it is safe to call {@link #build} multiple times to build
147   * multiple networks in series. Each new network contains all the elements of the ones created
148   * before it.
149   *
150   * @since 28.0
151   */
152  public static class Builder<N, E> {
153
154    private final MutableNetwork<N, E> mutableNetwork;
155
156    Builder(NetworkBuilder<N, E> networkBuilder) {
157      this.mutableNetwork = networkBuilder.build();
158    }
159
160    /**
161     * Adds {@code node} if it is not already present.
162     *
163     * <p><b>Nodes must be unique</b>, just as {@code Map} keys must be. They must also be non-null.
164     *
165     * @return this {@code Builder} object
166     */
167    @CanIgnoreReturnValue
168    public ImmutableNetwork.Builder<N, E> addNode(N node) {
169      mutableNetwork.addNode(node);
170      return this;
171    }
172
173    /**
174     * Adds {@code edge} connecting {@code nodeU} to {@code nodeV}.
175     *
176     * <p>If the network is directed, {@code edge} will be directed in this network; otherwise, it
177     * will be undirected.
178     *
179     * <p><b>{@code edge} must be unique to this network</b>, just as a {@code Map} key must be. It
180     * must also be non-null.
181     *
182     * <p>If {@code nodeU} and {@code nodeV} are not already present in this network, this method
183     * will silently {@link #addNode(Object) add} {@code nodeU} and {@code nodeV} to the network.
184     *
185     * <p>If {@code edge} already connects {@code nodeU} to {@code nodeV} (in the specified order if
186     * this network {@link #isDirected()}, else in any order), then this method will have no effect.
187     *
188     * @return this {@code Builder} object
189     * @throws IllegalArgumentException if {@code edge} already exists in the network and does not
190     *     connect {@code nodeU} to {@code nodeV}
191     * @throws IllegalArgumentException if the introduction of the edge would violate {@link
192     *     #allowsParallelEdges()} or {@link #allowsSelfLoops()}
193     */
194    @CanIgnoreReturnValue
195    public ImmutableNetwork.Builder<N, E> addEdge(N nodeU, N nodeV, E edge) {
196      mutableNetwork.addEdge(nodeU, nodeV, edge);
197      return this;
198    }
199
200    /**
201     * Adds {@code edge} connecting {@code endpoints}. In an undirected network, {@code edge} will
202     * also connect {@code nodeV} to {@code nodeU}.
203     *
204     * <p>If this network is directed, {@code edge} will be directed in this network; if it is
205     * undirected, {@code edge} will be undirected in this network.
206     *
207     * <p>If this network is directed, {@code endpoints} must be ordered.
208     *
209     * <p><b>{@code edge} must be unique to this network</b>, just as a {@code Map} key must be. It
210     * must also be non-null.
211     *
212     * <p>If either or both endpoints are not already present in this network, this method will
213     * silently {@link #addNode(Object) add} each missing endpoint to the network.
214     *
215     * <p>If {@code edge} already connects an endpoint pair equal to {@code endpoints}, then this
216     * method will have no effect.
217     *
218     * @return this {@code Builder} object
219     * @throws IllegalArgumentException if {@code edge} already exists in the network and connects
220     *     some other endpoint pair that is not equal to {@code endpoints}
221     * @throws IllegalArgumentException if the introduction of the edge would violate {@link
222     *     #allowsParallelEdges()} or {@link #allowsSelfLoops()}
223     * @throws IllegalArgumentException if the endpoints are unordered and the network is directed
224     */
225    @CanIgnoreReturnValue
226    public ImmutableNetwork.Builder<N, E> addEdge(EndpointPair<N> endpoints, E edge) {
227      mutableNetwork.addEdge(endpoints, edge);
228      return this;
229    }
230
231    /**
232     * Returns a newly-created {@code ImmutableNetwork} based on the contents of this {@code
233     * Builder}.
234     */
235    public ImmutableNetwork<N, E> build() {
236      return ImmutableNetwork.copyOf(mutableNetwork);
237    }
238  }
